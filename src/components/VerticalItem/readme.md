const containerStyles = {

    borderBottomLeftRadius: '0.875rem',

};

class SimpleVerticalNavigation extends React.Component {
    constructor(props) {

        this.state = {
            selectedItem: 'item 1',

        this.handleOnSelect = this.handleOnSelect.bind(this);

    handleOnSelect(e, selectedItem) {

}
